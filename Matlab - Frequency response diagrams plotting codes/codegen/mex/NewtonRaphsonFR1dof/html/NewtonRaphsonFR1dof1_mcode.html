1function [AmplitudeFR,AamplitudeFR,BamplitudeFR,PhaseAngleFR,W] = NewtonRaphsonFR1dof(tol,m,c,k,beta,fc,fMin,fMax,numAmplSamples,numPulsSamples,aMax,maxIter)
 2
 3
 4W = fMin*2*pi:((fMax-fMin)/(numPulsSamples-1))*2*pi:fMax*2*pi;
 5AmplitudeFR = zeros(numAmplSamples,length(W));
 6AamplitudeFR = zeros(numAmplSamples,length(W));
 7BamplitudeFR = zeros(numAmplSamples,length(W));
 8PhaseAngleFR = zeros(numAmplSamples,length(W));
 9
10for i=1:length(W)
11    for j=1:numAmplSamples
12        n = 1;
13        amplitude = [j*aMax/numAmplSamples; j*aMax/numAmplSamples];
14        F = FReq1dof(m,c,k,beta,fc,W(i),amplitude(1),amplitude(2));
15        while norm(F)>tol && n<=maxIter
16            F = FReq1dof(m,c,k,beta,fc,W(i),amplitude(1),amplitude(2));
17            J = FRjac1dof(m,c,k,beta,W(i),amplitude(1),amplitude(2));
18            amplitude = amplitude - J\F;
19            n = n+1;
20        end
21        if n<=maxIter,
22            AamplitudeFR(j,i) = amplitude(1);
23            BamplitudeFR(j,i) = amplitude(2);
24            AmplitudeFR(j,i) = sqrt(amplitude(2)^2+amplitude(1)^2);
25            PhaseAngleFR(j,i) = atan(abs(amplitude(2)/amplitude(1)));
26            if amplitude(1)>0, PhaseAngleFR(j,i) = -PhaseAngleFR(j,i);
27            elseif amplitude(1)<0, PhaseAngleFR(j,i) = PhaseAngleFR(j,i) - pi;
28            end
29        end
30    end
31end
32
33end
