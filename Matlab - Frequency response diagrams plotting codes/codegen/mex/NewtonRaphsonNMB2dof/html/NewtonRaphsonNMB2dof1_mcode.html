1function [Amplitude1NMB,Amplitude2NMB] = NewtonRaphsonNMB2dof(tol,m1,m2,k1,k2,beta,aMax,maxIter,W)
 2
 3
 4Amplitude1NMB = zeros(1,length(W));
 5Amplitude2NMB = zeros(1,length(W));
 6
 7for i=1:length(W)
 8    n = 1;
 9    amplitude = [aMax; -aMax];
10    F = NMBeq2dof(m1,m2,k1,k2,beta,W(i),amplitude(1),amplitude(2));
11    while norm(F)>tol && n<=maxIter
12        F = NMBeq2dof(m1,m2,k1,k2,beta,W(i),amplitude(1),amplitude(2));
13        J = NMBjac2dof(m1,m2,k1,k2,beta,W(i),amplitude(1),amplitude(2));
14        amplitude = amplitude - J\F;
15        n = n+1;
16    end
17    if n<=maxIter,
18        Amplitude1NMB(i) = amplitude(1);
19        Amplitude2NMB(i) = amplitude(2);
20    end
21end

23end
