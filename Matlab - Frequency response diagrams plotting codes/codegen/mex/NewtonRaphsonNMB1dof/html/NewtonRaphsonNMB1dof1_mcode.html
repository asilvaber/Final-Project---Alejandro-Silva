1function [AmplitudeNMB] = NewtonRaphsonNMB1dof(tol,m,k,beta,aMax,maxIter,W)
 2
 3
 4AmplitudeNMB = zeros(1,length(W));
 5
 6for i=1:length(W)
 7    n = 1;
 8    amplitude = aMax;
 9    F = NMBeq1dof(m,k,beta,W(i),amplitude);
10    while norm(F)>tol && n<=maxIter
11        F = NMBeq1dof(m,k,beta,W(i),amplitude);
12        J = NMBjac1dof(m,k,beta,W(i),amplitude);
13        amplitude = amplitude - J\F;

15    end
16    if n<=maxIter,

18    end
19end

21end
